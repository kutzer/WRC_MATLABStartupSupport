# Calculate the current time and the time 360 seconds ago
$currentDateTime = Get-Date
$startTime = $currentDateTime.AddSeconds(-360)

# Get files in the specified directory and its subdirectories with specified extensions created within the last 360 seconds

    ($_.CreationTime -ge $startTime) -and ($fileExtensions -contains $_.Extension)
}
